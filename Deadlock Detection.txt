#include <stdio.h>

int main() {
    int p=3, r=3;
    int alloc[3][3]={{0,1,0},{2,0,0},{3,0,2}};
    int avail[3]={0,0,0};
    int finish[3]={0};

    for(int k=0;k<p;k++){
        for(int i=0;i<p;i++){
            if(!finish[i]){
                int flag=1;
                for(int j=0;j<r;j++){
                    if(alloc[i][j]>avail[j])
                        flag=0;
                }
                if(flag){
                    finish[i]=1;
                    printf("P%d finished\n",i);
                }
            }
        }
    }
}
