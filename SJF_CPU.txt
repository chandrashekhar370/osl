#include <stdio.h>

int main() {
    int n, bt[10], wt[10], tat[10];

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(int i=0;i<n;i++){
        scanf("%d",&bt[i]);
    }

    // Sort
    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(bt[i]>bt[j]){
                int temp=bt[i];
                bt[i]=bt[j];
                bt[j]=temp;
            }
        }
    }

    wt[0]=0;
    for(int i=1;i<n;i++)
        wt[i]=wt[i-1]+bt[i-1];

    for(int i=0;i<n;i++){
        tat[i]=wt[i]+bt[i];
        printf("WT=%d TAT=%d\n",wt[i],tat[i]);
    }
}
